  void Motion_detector::send_message(){
    
    Message m = new Message(this->location_description);
    
    List<Guard> guards = property->get_guards();
    Guard g;
    
    g = guards.front();
    g.receiveMessage(m);
    
    int count = guards.size()-1;
    
    while(count > 0){
      g = guards.next();
      g.receiveMessage(m);
    }
